import hashlib, os
def find_dupes(root):
    seen = {}
    for d,_,fs in os.walk(root):
        for f in fs:
            p = os.path.join(d,f)
            h = hashlib.md5(open(p,"rb").read()).hexdigest()
            seen.setdefault(h,[]).append(p)
    return [v for v in seen.values() if len(v)>1]
